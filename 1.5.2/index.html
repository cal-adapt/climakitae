    ClimakitAE: Climate Data Analysis for California

    Query, process, and analyze downscaled climate projections for California with a modern Python API.

    ClimakitAE provides programmatic access to:
    - 🌡️ WRF dynamical downscaling — 3km, 9km, 45km resolution hourly to monthly data
    - 📊 LOCA2 statistical downscaling — Bias-corrected CMIP6 global models at regional scale
    - 🔄 Flexible processing — Spatial clipping, temporal subsetting, warming level analysis, unit conversion

    Built for climate scientists, environmental analysts, and Python developers.

    ⚡ Quick Start

    1. Install

    pip install climakitae

    2. Query Climate Data (5 minutes)

    from climakitae.new_core.user_interface import ClimateData

cd = ClimateData()

# Get WRF temperature for Los Angeles, 2015
data = (cd
    .catalog("cadcat")
    .activity_id("WRF")
    .institution_id("UCLA")
    .grid_label("d03")
    .table_id("mon")
    .variable("t2max")
    .processes({
        "time_slice": ("2015-01-01", "2015-12-31"),
        "clip": "Los Angeles",
        "convert_units": "degF"
    })
    .get())

data["t2max"].isel(sim=0).mean(dim="time").plot()

    🎯 What Can You Do?

    Analyze Temperature Projections

    # Compare warming at 2°C global warming across three California regions
cd = ClimateData()

regions = {"Bay Area": "San Francisco County", 
           "Central Valley": "Fresno County",
           "Southern CA": "Los Angeles County"}

for name, county in regions.items():
    data = (cd
        .catalog('cadcat')
        .activity_id('LOCA2')
        .variable("tasmax")
        .processes({
            "warming_level": {"warming_levels": [2.0]},
            "clip": county
        })
        .get())
    print(f"{name}: {data['tasmax'].mean().values:.1f}K")

    Build Energy Demand Models

    from climakitae.tools.derived_variables import compute_hdd_cdd

data = (cd
    .variable("tasmax")
    .processes({
        "convert_units": "degC",
        "time_slice": ("2030-01-01", "2060-12-31"),
        "clip": "Los Angeles"
    })
    .get())

hdd, cdd = compute_hdd_cdd(data["tasmax"], hdd_threshold=18.3, cdd_threshold=18.3)
annual_demand = (hdd + cdd).resample(time="YS").sum()

    Process Multi-Model Ensembles

    import pandas as pd
from climakitae.tools import batch_select

locations = pd.DataFrame({
    "name": ["LA", "SF", "SD"],
    "latitude": [34.05, 37.77, 32.72],
    "longitude": [-118.25, -122.42, -117.16]
})

results = batch_select(
    cd, 
    locations,
    variable="tasmax",
    activity_id="WRF",
    institution_id="UCLA"
)

# Multi-threaded processing across all locations
for loc, data in results.items():
    print(f"{loc}: {data['tasmax'].mean().compute().values:.1f}K")

    Export for External Tools

    # Export 35-year projection to cloud-optimized Zarr for QGIS analysis
data = (cd
    .variable("pr")
    .activity_id("LOCA2")
    .processes({
        "time_slice": ("2030-01-01", "2065-12-31"),
        "export": {
            "filename": "ca_precip_30year",
            "file_format": "Zarr",
            "mode": "local"
        }
    })
    .get())

    🚀 Key Features

    Flexible Querying

    # By named region
.processes({"clip": "Los Angeles"})

# By multiple locations
.processes({"clip": ["Alameda", "Contra Costa", "San Francisco Bay"]})

# By lat/lon point
.processes({"clip": (37.7749, -122.4194)})

# By bounding box
.processes({"clip": ((34.0, 36.0), (-121.0, -119.0))})

    Climate-Aware Subsetting

    # Calendar-based
.processes({"time_slice": ("2030-01-01", "2060-12-31")})

# Warming level-based (more meaningful for policy)
.processes({"warming_level": {"warming_levels": [1.5, 2.0, 3.0]}})

    Production-Ready Export

    • NetCDF — CF conventions, full metadata
    • Zarr — Cloud-optimized, scalable for large datasets
    • CSV — Tabular time series for spreadsheet tools
    • GeoTIFF — Raster for GIS software (QGIS, ArcGIS)

    Lazy Evaluation

    # Data isn't loaded until you compute
data = cd.variable("tasmax").get()  # Builds query graph
result = data.mean().compute()      # Executes here
