class Boundaries:
"""
Manages geographic boundary data with lazy loading from external catalogs.
This class provides access to US state boundaries while handling potential
failures from external data sources like network issues or permissions.

Parameters
----------
boundary_catalog: External catalog containing boundary data sources

Attributes
----------
_us_states: Lazy-loaded property for US states boundary data.

Returns
-------
The US states boundary data from the external catalog
Note:
Data is loaded only once and cached. May raise exceptions if
external catalog access fails due to network or permission issues.
"""
def __init__(self, boundary_catalog):
self._cat = boundary_catalog # üö® External dependency!

@property
def _us_states(self):
if self.__us_states is None:
# Could fail: network, permissions, etc.
self.__us_states = self._cat.states.read()
return self.__us_states
üö® Problem: External dependencies can introduce uncertainty into your tests.
üéØ Solution: Mock external dependencies, test YOUR logic!

Mocking: Do's and Don'ts

‚úÖ Good Mocking

# Mock external dependencies only
@patch('climakitae.core.boundaries.intake.open_catalog')
def test_boundaries_init(self, mock_catalog):
mock_catalog.return_value.states.read.return_value = test_df
# Test OUR logic with controlled inputs
boundaries = Boundaries(mock_catalog)
assert boundaries._us_states is not None

‚ùå Over-Mocking

# DON'T mock your own methods!
@patch('boundaries._process_ca_counties')
@patch('boundaries._sort_by_name')
@patch('boundaries._validate_data')
def test_overmocked(m1, m2, m3):
m1.return_value = "fake_result"
# This tests mocks, not real code! üò±
result = boundaries.process_counties()
assert result == "fake_result" # Meaningless!
üéØ Rule: Mock dependencies (external APIs, file systems, databases), not your own code!

üèóÔ∏è Fixtures: Reusable Test Setup

@pytest.fixture
def mock_boundaries(self):
"""Create Boundaries with mocked catalog."""
mock_catalog = MagicMock()
for attr in ["states", "counties", "huc8"]:
catalog_entry = MagicMock()
catalog_entry.read.return_value = self._create_mock_dataframe(attr)
setattr(mock_catalog, attr, catalog_entry)

yield Boundaries(mock_catalog)
# Teardown if needed
üéØ Benefits of MagicMock:
‚Ä¢ Auto-creates missing attributes (no AttributeError)
‚Ä¢ Supports magic methods (__len__, __str__)
‚Ä¢ Cleaner than basic Mock for complex objects

üéØ Benefits of yield:
‚Ä¢ Automatic cleanup after test completion
‚Ä¢ Fixture runs setup ‚Üí test ‚Üí teardown
‚Ä¢ Resource management (files, connections)
‚Ä¢ Exception handling

üéØ Patching: Where and How

‚úÖ Patch WHERE It's Imported

# In user_interface.py:
from climakitae.new_core.dataset_factory import DatasetFactory

class ClimateData:
def __init__(self):
self.factory = DatasetFactory()

# ‚úÖ CORRECT: Patch at import location
@patch('climakitae.new_core.user_interface.DatasetFactory')
def test_climate_data(mock_factory):
climate_data = ClimateData()

‚ùå Don't Patch Source Location

# ‚ùå WRONG: Patching where it's defined
@patch('climakitae.new_core.dataset_factory.DatasetFactory')
def test_climate_data(mock_factory):
# This won't work! The import already happened
climate_data = ClimateData()

# ‚ùå WRONG: Patching built-ins incorrectly
@patch('builtins.open') # Too broad!
def test_file_read(mock_open):
# Affects ALL file operations

üéØ Context Manager Patching

# Multiple patches together
def test_complex_initialization(self):
with patch('module.DatasetFactory') as mock_factory, \
patch('module.read_csv_file') as mock_csv, \
patch('builtins.print') as mock_print:

mock_csv.return_value = pd.DataFrame()
obj = ClimateData()
assert mock_factory.called

üé® Decorator Patching

# Single patch for entire method
@patch('module.DatasetFactory')
def test_simple_case(self, mock_factory):
mock_factory.return_value.create.return_value = "data"
obj = ClimateData()
assert obj.get_data() == "data"

# Or apply to entire class
@patch('module.external_service')
class TestClimateData:
def test_method1(self, mock_service):
pass
üéØ When to Use Each Approach:
‚Ä¢ Context Managers: Multiple patches, conditional patching, limited scope
‚Ä¢ Decorators: Single patch, entire method/class scope, cleaner syntax

üéØ What Properties Should You Test?

test_categories = {
"happy_path": "üòä Normal, expected usage patterns",
"edge_cases": "üîç Empty data, boundaries, limits",
"error_conditions": "üí• Invalid input, network failures",
"state_changes": "üîÑ Before/after behavior",
"performance": "‚ö° Memory usage, timing constraints",
"integration": "üîó Components working together"
}

‚úÖ Test These Behaviors

  • Return Values: Correct data types & content
  • State Changes: Object properties after operations
  • Side Effects: Files created, APIs called
  • Error Handling: Graceful failure modes
  • Boundary Conditions: Min/max values
  • Caching: Data retrieved once, reused

‚ùå Don't Test These Details

  • Internal Methods: Private implementation
  • Third-party Code: Libraries you don't own
  • Framework Magic: Django ORM internals
  • Trivial Code: Simple getters/setters
  • Constants: Static configuration values
  • External APIs: Mock them instead!
üí° Golden Rule: Test behavior, not implementation! Ask "What should this do?" not "How does it work?"

üö® Common Testing Pitfalls

‚ùå Don't Mock What You're Testing

# BAD: Mocking our own method!
@patch('boundaries._process_ca_counties')
def test_ca_counties(mock_process):
mock_process.return_value = expected
result = boundaries._process_ca_counties(data)
# This tests nothing! üò±

‚ùå Brittle Test Assertions

# BAD: Brittle - breaks when data changes
assert len(boundary_dict) == 8 # Breaks when we add datasets
assert list(result.keys()) == ["none", "lat/lon"...]

# GOOD: Test behavior, not exact counts
assert "CA counties" in boundary_dict
assert isinstance(result["states"], dict)
